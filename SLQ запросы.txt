PRAGMA table_info(Shops);

===== Создание таблицы (простая) =====

CREATE TABLE Shops (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT,
    url TEXT,
    category TEXT,
    location TEXT,
    icon BLOB
);

===== Создание таблицы (один к многим) =====

CREATE TABLE Sellers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    id_shop INTEGER NOT NULL,
    name TEXT,
    url TEXT,
    category TEXT,
    icon BLOB,
    FOREIGN KEY(id_shop) REFERENCES Shops(id)
);

===== Обновление =====
